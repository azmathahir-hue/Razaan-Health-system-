    // ☁️ FIREBASE CLOUD SYNC METHOD
    function submitToCloud() {
        const id = document.getElementById('pId').value.trim();
        if(!id) { alert("Fadlan kowaad geli Patient ID qaybta Diiwaangelinta!"); switchModule('demo'); return; }

        let vitals = [];
        document.querySelectorAll('#vitalsMasterWrapper .log-sheet-card').forEach(card => {
            vitals.push({
                date: card.querySelector('.v-date').value,
                bp: [card.querySelector('.v-bp-m').value, card.querySelector('.v-bp-md').value, card.querySelector('.v-bp-n').value],
                hr: [card.querySelector('.v-hr-m').value, card.querySelector('.v-hr-md').value, card.querySelector('.v-hr-n').value],
                rr: [card.querySelector('.v-rr-m').value, card.querySelector('.v-rr-md').value, card.querySelector('.v-rr-n').value],
                bs: [card.querySelector('.v-bs-m').value, card.querySelector('.v-bs-md').value, card.querySelector('.v-bs-n').value],
                o2: [card.querySelector('.v-o2-m').value, card.querySelector('.v-o2-md').value, card.querySelector('.v-o2-n').value],
                temp: [card.querySelector('.v-temp-m').value, card.querySelector('.v-temp-md').value, card.querySelector('.v-temp-n').value]
            });
        });

        let progress = [];
        document.querySelectorAll('#progressTable tbody tr').forEach(r => {
            progress.push({ date: r.querySelector('.d-date').value, time: r.querySelector('.d-time').value, note: r.querySelector('.d-note').value });
        });

        let meds = [];
        document.querySelectorAll('#medsTable tbody tr').forEach(r => {
            meds.push({ name: r.querySelector('.m-name').value, dose: r.querySelector('.m-dose').value, freq: r.querySelector('.m-freq').value, stopDate: r.querySelector('.m-stop').value });
        });

        const finalData = {
            patientId: id, fullName: document.getElementById('pName').value,
            age: document.getElementById('pAge').value, phone: document.getElementById('pPhone').value,
            vitalsHistory: vitals, progressNotes: progress, medications: meds,
            updatedAt: new Date().toISOString()
        };

        db.collection("bukaannada_razaan").doc(id).set(finalData)
        .then(() => alert("Dhamaan xogta shaxda " + id + " si sax ah ayaa loogu kaydiyey daruuraha Firestore! 🚀"))
        .catch(err => alert("Cillad dhacday: " + err.message));
    }

    // 🔍 LOOKUP PATIENT FROM FIRESTORE CLOUD
    function lookupChart() {
        const sid = document.getElementById('srcId').value.trim();
        const box = document.getElementById('searchResultBox');
        if(!sid) return;

        db.collection("bukaannada_razaan").doc(sid).get().then(doc => {
            box.style.display = "block";
            if(doc.exists) {
                const d = doc.data();
                box.style.backgroundColor = "#ecfdf5"; box.style.color = "#065f46"; box.style.border = "1px solid #10b981";
                box.innerHTML = `<strong>Bukaan La Helay:</strong> ${d.fullName}<br>• Vitals Sheets: ${d.vitalsHistory ? d.vitalsHistory.length : 0} bog.<br>• Progress Records: ${d.progressNotes ? d.progressNotes.length : 0} saf.<br>• Dawooyinka firfircoon: ${d.medications ? d.medications.length : 0} xabbo.`;

        });
    }
